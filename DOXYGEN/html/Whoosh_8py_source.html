95def index_documents(index_dir, json_file, batch_size=1000):
96 """!
97 @brief Index documents from JSON file into Whoosh index
98 @param index_dir Path to the index directory
99 @param json_file Path to the JSON document file
100 @param batch_size Number of documents to process in each batch (default: 1000)
101 @return Index object for the populated index
102 @details Forces rebuild of index, processes documents in batches for memory efficiency.
103 Uses incremental JSON parsing to handle large files.
104 """
105 print(f"--- Starting Whoosh Indexing ---")
106 print(f"Index directory: {index_dir}")
107 print(f"JSON source file: {json_file}")
108 # Crea un nuovo indice (forza la ricreazione)
109 ix = create_whoosh_index_internal(index_dir)
110 doc_count = 0
111
112 try:
113 # Leggere il file JSON in modo incrementale con ijson
114 with open(json_file, 'r', encoding='utf-8', errors='ignore') as f:
115 parser = ijson.parse(f)
116 current_doc = {}
117 batch = []
118
119 for prefix, event, value in parser:
120 if prefix.endswith('.id'):
121 if current_doc: # Se abbiamo un documento completo, aggiungiamolo al batch
122 batch.append(current_doc)
123 current_doc = {}
124 current_doc['id'] = str(value)
125 elif prefix.endswith('.title'):
126 current_doc['title'] = value
127 elif prefix.endswith('.abstract'):
128 current_doc['abstract'] = value
129 elif prefix.endswith('.corpus'):
130 current_doc['corpus'] = value
131 elif prefix.endswith('.keywords'):
132 current_doc['keywords'] = value
133 elif prefix.endswith('.url'):
134 current_doc['url'] = value
135
136 # Se il batch raggiunge la dimensione specificata, scrivilo nell'indice
137 if len(batch) >= batch_size:
138 with ix.writer() as writer:
139 for doc in batch:
140 writer.add_document(
141 id=doc['id'],
142 title=doc.get('title', ''),
143 abstract=doc.get('abstract', ''),
144 corpus=doc.get('corpus', ''),
145 keywords=doc.get('keywords', ''),
146 url=doc.get('url', '')
147 )
148 doc_count += len(batch)
149 print(f"üì¶ Batch di {len(batch)} documenti indicizzati. Totale: {doc_count}")
150 batch = []
151
152 # Gestisci l'ultimo documento e batch
153 if current_doc:
154 batch.append(current_doc)
155 if batch:
156 with ix.writer() as writer:
157 for doc in batch:
158 writer.add_document(
159 id=doc['id'],
160 title=doc.get('title', ''),
161 abstract=doc.get('abstract', ''),
162 corpus=doc.get('corpus', ''),
163 keywords=doc.get('keywords', ''),
164 url=doc.get('url', '')
165 )
166 doc_count += len(batch)
167 print(f"üì¶ Ultimo batch di {len(batch)} documenti indicizzati. Totale finale: {doc_count}")
168
169 except Exception as e:
170 print(f"‚ùå Errore durante l'indicizzazione: {e}")
171 traceback.print_exc() # Print full traceback
172 # raise # Re-raise if you want to stop execution

174 print(f"‚úÖ Indicizzazione completata! Totale documenti indicizzati: {doc_count}")

178def parse_advanced_query(query_string, schema):
179 """!
180 @brief Parse complex query strings with field specifications and operators
181 @param query_string Query string to parse (may contain field:term AND/OR operators)
182 @param schema Whoosh schema object for field validation
183 @return Parsed Query object or None if parsing fails
184 @details Handles field-specific searches, phrase queries, and logical operators.
185 Expands non-field queries to search across multiple fields.
186 """
187 try:
188 # Check if query has any field specifications
189 has_field_specs = any(field + ":" in query_string for field in ['title', 'abstract', 'corpus', 'keywords'])
190
191 # For queries without field specifications but with AND/OR operators
192 if not has_field_specs and (" AND " in query_string or " OR " in query_string):
193 print(f"Processing query without field specs: {query_string}")
194
195 # Create a specific parser for non-field queries with AND/OR
196 parser = qparser.QueryParser("content", schema)
197 parser.add_plugin(qparser.GroupPlugin())
198 parser.add_plugin(qparser.PhrasePlugin())
199
200 # Split the query into terms respecting quotes
201 terms = []
202 current_term = ""
203 in_quotes = False
204
205 for char in query_string:
206 if char == '"':
207 in_quotes = not in_quotes
208 current_term += char
209 elif char == ' ' and not in_quotes:
210 if current_term:
211 terms.append(current_term)
212 current_term = ""
213 else:
214 current_term += char
215
216 if current_term:
217 terms.append(current_term)
218
219 # Build the query with appropriate operators
220 final_query_parts = []
221 i = 0
222 while i < len(terms):
223 if terms[i].upper() in ["AND", "OR"]:
224 # Skip operators
225 i += 1
226 continue
227
228 # Handle phrase (with quotes)
229 term = terms[i].strip('"')
230
231 # Build a sub-query that searches this term in all fields
232 sub_query = []
233 for field in ['title', 'abstract', 'corpus']:
234 if '"' in terms[i]:
235 # For phrases, keep them intact
236 sub_query.append(f'{field}:"{term}"')
237 else:
238 # For single terms
239 sub_query.append(f'{field}:{term}')
240
241 final_query_parts.append("(" + " OR ".join(sub_query) + ")")
242
243 # Check if we need to add an operator
244 if i + 1 < len(terms) and terms[i+1].upper() in ["AND", "OR"]:
245 final_query_parts.append(terms[i+1])
246 i += 2
247 else:
248 i += 1
249
250 # Join all parts
251 expanded_query = " ".join(final_query_parts)
252 print(f"Expanded query: {expanded_query}")
253
254 # Parse with a field-aware parser
255 field_parser = qparser.QueryParser("content", schema)
256 field_parser.add_plugin(qparser.FieldsPlugin())
257 field_parser.add_plugin(qparser.PhrasePlugin())
258 field_parser.add_plugin(qparser.OperatorsPlugin())
259
260 return field_parser.parse(expanded_query)
261
262 # For field-specific queries or simple queries
263 elif has_field_specs:
264 # Use a parser that respects field prefixes
265 parser = qparser.QueryParser("content", schema)
266 parser.add_plugin(qparser.FieldsPlugin())
267 else:
268 # For simple queries without operators, search across all fields
269 parser = qparser.MultifieldParser(['title', 'abstract', 'corpus'],
270 schema,
271 group=qparser.OrGroup)
272
273 # Add plugins for advanced features
274 parser.add_plugin(qparser.PhrasePlugin())
275 parser.add_plugin(qparser.OperatorsPlugin())
276
277 # Parse and return the query
278 return parser.parse(query_string)
279
280 except Exception as e:
281 print(f"Error parsing advanced query: {e}")
282 print(f"Query string was: {query_string}")
283 import traceback
284 traceback.print_exc()
285 return None

457def process_natural_query(query_string):
458 """!
459 @brief Process natural language query with NLP techniques
460 @param query_string The natural language query to process
461 @return Processed query string with expanded terms
462 @details Uses YAKE keyword extraction, lemmatization, and WordNet expansion
463 for synonyms, hypernyms, and hyponyms. Includes fallback mechanisms.
464 """
465 # Inizializza gli strumenti NLP
466 lemmatizer = WordNetLemmatizer()
467 stop_words = set(stopwords.words('english'))
468
469 # 1. Keyword extraction with YAKE
470 kw_extractor = yake.KeywordExtractor(
471 lan="en",
472 n=3,
473 dedupLim=0.9,
474 dedupFunc='seqm',
475 windowsSize=1,
476 top=5, # Estrai le top 5 keyword
477 features=None
478 )
479 keywords = [kw[0].lower() for kw in kw_extractor.extract_keywords(query_string)]
480
481 # Tokenizzazione e rimozione stopwords dalla query originale
482 tokens = word_tokenize(query_string.lower())
483
484 # Combina keyword e token originali per la lemmatizzazione e l'espansione
485 # per assicurarsi che le keyword multi-parola siano considerate
486 base_terms_for_nlp = list(set(keywords + tokens))
487 base_terms_for_nlp = [term for term in base_terms_for_nlp if term.isalnum() and term not in stop_words]
488
489 pos_tags = pos_tag(base_terms_for_nlp)
490
491 lemmatized = [lemmatizer.lemmatize(word, get_wordnet_pos(tag))
492 for word, tag in pos_tags]
493
494 # Espansione con sinonimi, iperonimi e iponimi
495 expanded_terms = set()
496 # Aggiungi le keyword estratte da YAKE (gi√† in minuscolo) e i termini lemmatizzati originali
497 expanded_terms.update(keywords)
498 expanded_terms.update(lemmatized)
499
500 # Termini su cui basare l'espansione WordNet (keyword + lemmi originali)
501 terms_for_wordnet_expansion = set(keywords + lemmatized)
502
503 for term in terms_for_wordnet_expansion:
504 # Aggiungi il termine stesso (gi√† fatto, ma per sicurezza)
505 expanded_terms.add(term)
506 synsets = wordnet.synsets(term)
507 for synset in synsets[:2]: # Limita a 2 synset per termine per evitare rumore
508 # Aggiungi sinonimi
509 expanded_terms.update(lemma.name().lower().replace('_', ' ') for lemma in synset.lemmas() if lemma.name().lower() not in stop_words)
510
511 # Aggiungi iperonimi
512 for hypernym in synset.hypernyms():
513 expanded_terms.update(lemma.name().lower().replace('_', ' ') for lemma in hypernym.lemmas() if lemma.name().lower() not in stop_words)
514
515 # Aggiungi iponimi (termini pi√π specifici)
516 for hyponym in synset.hyponyms()[:1]: # Limita a 1 iponimo per synset per non allargare eccessivamente
517 expanded_terms.update(lemma.name().lower().replace('_', ' ') for lemma in hyponym.lemmas() if lemma.name().lower() not in stop_words)
518
519 # Pulisci i termini finali
520 final_expanded_terms = {
521 term.strip() for term in expanded_terms if term.strip() and term.strip() not in stop_words and len(term.strip()) > 1
522 }
523
524 if not final_expanded_terms: # Fallback se nessun termine √® stato generato
525 # Usa le keyword o i lemmi originali se disponibili
526 fallback_terms = set(keywords + lemmatized)
527 fallback_terms = {t for t in fallback_terms if t and t not in stop_words and len(t) > 1}
528 if fallback_terms:
529 processed_query = ' OR '.join(fallback_terms)
530 else: # Fallback estremo: usa la query originale tokenizzata e senza stopwords
531 original_tokens_no_stops = [token for token in word_tokenize(query_string.lower()) if token.isalnum() and token not in stop_words and len(token)>1]
532 if original_tokens_no_stops:
533 processed_query = ' OR '.join(original_tokens_no_stops)
534 else:
535 processed_query = query_string # Ultimo fallback, query originale
536 else:
537 processed_query = ' OR '.join(final_expanded_terms)
538
539 print(f"Whoosh - Query elaborata: {processed_query}")
540 return processed_query
