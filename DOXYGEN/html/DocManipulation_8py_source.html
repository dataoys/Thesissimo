28def cleanText(text):
29 """!
30 @brief Comprehensive text cleaning for document content
31 @param text Raw text content to clean
32 @return Cleaned and normalized text
33 @details Performs multiple cleaning operations:
34 - LaTeX expression removal
35 - Control character filtering
36 - Whitespace normalization
37 - Newline cleanup
38 """
39 # Prima rimuovi il LaTeX
40 text = remove_latex(text)
41
42 # Pulizia base
43 text = re.sub(r'\s+', ' ', text)
44 text = ''.join(char for char in text if char.isprintable())
45
46 # Pulizie aggiuntive
47 text = re.sub(r'\n+', '\n', text) # Rimuove linee vuote multiple
48 text = re.sub(r'[\t\r\f\v]', '', text) # Rimuove altri caratteri di spaziatura
49 text = re.sub(r'\s*\n\s*', '\n', text) # Pulisce gli spazi intorno ai newline
50 text = re.sub(r' +', ' ', text) # Rimuove spazi multipli
51
52 return text.strip()
