67def plot_precision_recall_metrics(engine_metrics, engine_name, k_values):
68 """!
69 @brief Generate and save precision-recall plots for an engine using Seaborn
70 @param engine_metrics Dictionary containing metrics data for the engine
71 @param engine_name Name of the search engine (PyLucene, Whoosh, PostgreSQL)
72 @param k_values List of k values used for P@k and R@k calculations
73 @details Creates box plots, bar plots, histograms and scatter plots for:
74 - Precision@k distribution and means
75 - Recall@k distribution and means
76 - Average Precision distribution
77 - Response time analysis
78 @return None
79 """
80 # Determine plot directory based on engine name
81 if engine_name.lower() == "pylucene":
82 plot_dir = PYLUCENE_PLOTS_DIR
83 elif engine_name.lower() == "whoosh":
84 plot_dir = WHOOSH_PLOTS_DIR
85 elif engine_name.lower() == "postgresql":
86 plot_dir = POSTGRES_PLOTS_DIR
87 else:
88 plot_dir = PLOTS_DIR / engine_name.lower()
89 plot_dir.mkdir(parents=True, exist_ok=True)
90
91 # Set seaborn style
92 sns.set_style("whitegrid")
93 sns.set_palette("husl")
94
95 # 1. Plot Precision@k for all k values
96 plt.figure(figsize=(12, 8))
97
98 # Prepare data for precision plot
99 precision_data = []
100 for k in k_values:
101 for query_idx, precision in enumerate(engine_metrics["P@k"][k]):
102 precision_data.append({
103 'Query': f'Q{query_idx+1}',
104 'K': f'P@{k}',
105 'Precision': precision
106 })
107
108 if precision_data:
109 import pandas as pd
110 df_precision = pd.DataFrame(precision_data)
111
112 plt.subplot(2, 2, 1)
113 sns.boxplot(data=df_precision, x='K', y='Precision')
114 plt.title(f'{engine_name} - Precision@k Distribution')
115 plt.ylabel('Precision')
116 plt.ylim(0, 1)
117
118 plt.subplot(2, 2, 2)
119 sns.barplot(data=df_precision, x='K', y='Precision', estimator=np.mean, ci=95)
120 plt.title(f'{engine_name} - Mean Precision@k')
121 plt.ylabel('Mean Precision')
122 plt.ylim(0, 1)
123
124 # 2. Plot Recall@k for all k values
125 recall_data = []
126 for k in k_values:
127 for query_idx, recall in enumerate(engine_metrics["R@k"][k]):
128 recall_data.append({
129 'Query': f'Q{query_idx+1}',
130 'K': f'R@{k}',
131 'Recall': recall
132 })
133
134 if recall_data:
135 df_recall = pd.DataFrame(recall_data)
136
137 plt.subplot(2, 2, 3)
138 sns.boxplot(data=df_recall, x='K', y='Recall')
139 plt.title(f'{engine_name} - Recall@k Distribution')
140 plt.ylabel('Recall')
141 plt.ylim(0, 1)
142
143 plt.subplot(2, 2, 4)
144 sns.barplot(data=df_recall, x='K', y='Recall', estimator=np.mean, ci=95)
145 plt.title(f'{engine_name} - Mean Recall@k')
146 plt.ylabel('Mean Recall')
147 plt.ylim(0, 1)
148
149 plt.tight_layout()
150 plt.savefig(plot_dir / f'{engine_name}_precision_recall_at_k.png', dpi=300, bbox_inches='tight')
151 plt.close()
152
153 # 3. Plot Average Precision (AP) distribution
154 plt.figure(figsize=(10, 6))
155
156 if engine_metrics["AP"]:
157 ap_data = pd.DataFrame({
158 'Query': [f'Q{i+1}' for i in range(len(engine_metrics["AP"]))],

162 plt.subplot(1, 2, 1)

168 plt.subplot(1, 2, 2)
169 sns.boxplot(data=ap_data, y='Average_Precision')
170 plt.title(f'{engine_name} - AP Box Plot')
171 plt.ylabel('Average Precision')
172 plt.ylim(0, 1)
173
174 plt.tight_layout()
175 plt.savefig(plot_dir / f'{engine_name}_average_precision.png', dpi=300, bbox_inches='tight')
176 plt.close()
177
178 # 4. Plot Response Times
179 plt.figure(figsize=(10, 6))
180
181 if engine_metrics["ResponseTimes"]:
182 response_data = pd.DataFrame({
183 'Query': [f'Q{i+1}' for i in range(len(engine_metrics["ResponseTimes"]))],
184 'Response_Time': engine_metrics["ResponseTimes"]

186
187 plt.subplot(1, 2, 1)
188 sns.histplot(data=response_data, x='Response_Time', bins=20, kde=True)
189 plt.title(f'{engine_name} - Response Time Distribution')
190 plt.xlabel('Response Time (seconds)')
191
192 plt.subplot(1, 2, 2)
193 sns.scatterplot(data=response_data, x='Query', y='Response_Time')
