20def extract_keywords(uin):
21 """!
22 @brief Extract keywords from user input using NLTK POS tagging
23 @param uin Search bar input string
24 @return List of keywords filtered from nouns and verbs
25 @details Tokenizes input, performs POS tagging, and filters for meaningful
26 parts of speech while removing English stopwords
27 """
28
29 # Tokenizza la stringa di input
30 tokens = word_tokenize(uin)
31
32 # Esegui il POS tagging per identificare il tipo di parola (sostantivo, verbo, ecc.)
33 tagged = pos_tag(tokens)
34
35 # Estrai solo sostantivi e verbi (per ora ignoriamo le stopwords)
36 keywords = [word for word, tag in tagged if tag in ['NN', 'VB', 'NNS', 'VBD', 'VBG', 'VBN', 'VBZ']]
37
38 # Filtra le parole chiave rimuovendo le stopwords
39 stop_words = set(stopwords.words('english'))
40 filtered_keywords = [word for word in keywords if word.lower() not in stop_words]
41
42 return filtered_keywords

44def parse_advanced_query(query_string):
45 """!
46 @brief Parse field-specific query strings with AND/OR operators
47 @param query_string Query string with field specifications (e.g., "title:space AND corpus:python")
48 @return Tuple of (where_clause, params, fields_used)
49 @details Supports multi-word searches in quotes and logical operators.
50 Returns SQL WHERE clause, parameters, and field usage tracking.
51 """
52 parts = query_string.split(' AND ')
53 query_parts = []
54 params = []
55 fields_used = {'title': False, 'abstract': False, 'corpus': False}
56
57 for part in parts:
58 if ' OR ' in part:
59 or_parts = part.split(' OR ')
60 or_query_parts = []
61
62 for or_part in or_parts:
63 if ':' in or_part:
64 field, term = or_part.split(':', 1)
65 field = field.lower().strip()
66 term = term.strip().strip('"').strip("'").strip()
67
68 if field in fields_used:
69 fields_used[field] = True
70 or_query_parts.append(f"{field}_tsv @@ phraseto_tsquery('english', %s)")
71 params.append(term)
72
73 if or_query_parts:
74 query_parts.append('(' + ' OR '.join(or_query_parts) + ')')
75 else:
76 # Gestione diretta dei termini AND senza OR
77 if ':' in part:
78 field, term = part.split(':', 1)
79 field = field.lower().strip()
80 term = term.strip().strip('"').strip("'").strip()
81
82 if field in fields_used:
83 fields_used[field] = True
84 query_parts.append(f"{field}_tsv @@ phraseto_tsquery('english', %s)")
85 params.append(term)
86 else:
87 # Se nessun campo √® specificato, cerca in tutti i campi usando phraseto_tsquery
88 part = part.strip().strip('"').strip("'").strip()
89 all_fields_query = []
90 for field in fields_used:
91 fields_used[field] = True
92 all_fields_query.append(f"{field}_tsv @@ phraseto_tsquery('english', %s)")
93 params.append(part)
94 if all_fields_query:
95 query_parts.append('(' + ' OR '.join(all_fields_query) + ')')
96
97
98 if query_parts:
99 # Combina tutte le parti con AND
100 return ' AND '.join(query_parts), params, fields_used
101 return None, [], fields_used

103def build_search_query(search_terms, title_true, abstract_true, corpus_true, ranking_type='ts_rank_cd'):
104 """!
105 @brief Build PostgreSQL full-text search query with ranking
106 @param search_terms The search terms to look for
107 @param title_true Boolean flag to search in title field
108 @param abstract_true Boolean flag to search in abstract field
109 @param corpus_true Boolean flag to search in corpus field
110 @param ranking_type The ranking function to use ('ts_rank' or 'ts_rank_cd')
111 @return Tuple of (query_string, params) for PostgreSQL execution
112 @details Constructs SQL query with tsvector matching and relevance ranking
113 """
114 # Build the ranking expression based on selected fields
115 rank_fields = []
116 where_conditions = []
117 if title_true:
118 rank_fields.append('title_tsv')
119 where_conditions.append('title_tsv @@ to_tsquery(\'english\', %s)')
120 if abstract_true:
121 rank_fields.append('abstract_tsv')
122 where_conditions.append('abstract_tsv @@ to_tsquery(\'english\', %s)')
123 if corpus_true:
124 rank_fields.append('corpus_tsv')
125 where_conditions.append('corpus_tsv @@ to_tsquery(\'english\', %s)')
126
127 if not rank_fields:
128 return None, []
129
130 rank_expr = ' || '.join(rank_fields)
131 where_clause = ' OR '.join(where_conditions)
132
133 query = f"""
134 SELECT id, title, abstract, corpus, keywords, url,
135 {ranking_type}({rank_expr}, to_tsquery('english', %s)) as rank
136 FROM docs
137 WHERE {where_clause}
138 ORDER BY rank DESC
139 LIMIT 100
140 """
141
142 # Parameters for the query - one for ranking and one for each where condition
143 params = [search_terms] * (len(where_conditions) + 1)
144 return query, params

146def search(search_query, title_true, abstract_true, corpus_true, ranking_type='ts_rank_cd'):
147 """!
148 @brief Main PostgreSQL search function with dual search modes
149 @param search_query The user's input query string
150 @param title_true Boolean flag to search in title field
151 @param abstract_true Boolean flag to search in abstract field
152 @param corpus_true Boolean flag to search in corpus field
153 @param ranking_type The ranking function to use ('ts_rank' or 'ts_rank_cd')
154 @return List of matching documents ordered by relevance
155 @details Supports both field-specific searches (with AND/OR syntax) and
156 checkbox-based searches. Uses PostgreSQL full-text search capabilities.
157 @throws ValueError If database connection fails or query execution errors occur
158 """
159 conn = dbConn()
160 cur = conn.cursor()
161
162 try:
163 # Check if the query contains field-specific searches
164 if 'AND' in search_query or 'OR' in search_query:
165 where_clause, params, fields_used = parse_advanced_query(search_query)
166 if where_clause:
167 # Override checkbox selections with fields explicitly used in the query
168 title_true = title_true or fields_used['title']
169 abstract_true = abstract_true or fields_used['abstract']
170 corpus_true = corpus_true or fields_used['corpus']
171
172 # Build query with both explicit field searches and checkbox selections

174 SELECT id, title, abstract, corpus, keywords, url,
175 {ranking_type}(
176 CASE WHEN {title_true} THEN title_tsv ELSE ''::tsvector END ||
177 CASE WHEN {abstract_true} THEN abstract_tsv ELSE ''::tsvector END ||
178 CASE WHEN {corpus_true} THEN corpus_tsv ELSE ''::tsvector END,
179 phraseto_tsquery('english', %s)
180 ) as rank
181 FROM docs
182 WHERE {where_clause}
183 ORDER BY rank DESC
184 LIMIT 100
185 '''
186 # Add parameter for ranking calculation
187 if params:
188 execution_params = [params[0]] + params
189 else:
190 execution_params = []
191
192 # DEBUG PRINTS:

194 print(f"DEBUG: Parsed where_clause: {where_clause}")
