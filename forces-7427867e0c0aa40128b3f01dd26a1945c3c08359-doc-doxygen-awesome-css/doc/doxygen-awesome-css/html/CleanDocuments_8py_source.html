23def clean_mathematical_text(text):
24 """!
25 @brief Clean text containing mathematical expressions and Unicode artifacts
26 @param text Input text with mathematical notation and Unicode characters
27 @return Cleaned text with mathematical expressions normalized
28 @details Removes specific Unicode characters, LaTeX artifacts, HTML conversion errors,
29 and mathematical notation while preserving readable content
30 """
31 if not isinstance(text, str):
32 return text
33
34 # Remove HTML conversion error message with flexible whitespace matching
35 error_pattern = r'HTML conversions sometimes display errors.*?best practices\.'
36 text = re.sub(error_pattern, '', text, flags=re.IGNORECASE | re.DOTALL)
37
38 # Remove any remaining artifacts from the error message
39 text = re.sub(r'Authors:.*?supported packages\.', '', text, flags=re.IGNORECASE | re.DOTALL)
40 # Remove specific unicode characters
41 replacements = {
42 'italic_': '',
43 'bold_': '',
44 'blackboard_': '',
45 'start_POSTSUBSCRIPT': '_',
46 'end_POSTSUBSCRIPT': '',
47 'start_POSTSUPERSCRIPT': '^',
48 'end_POSTSUPERSCRIPT': '',
49 'start_ARG': '',
50 'end_ARG': '',
51 'divide': '/',
52 '\u011f': '',
53 '\ufffd': '',
54 '\u2018': '',
55 '\u20ac': '',
56 '\u00e2': '',
57 '\u02c6': '',
58 '\u0152': '',
59 '\u201e': '',
60 '\u00b4': '',
61 '\u2013': '',
62 '\u2014': '',
63 '\u203a': '',
64 '\u0178': '',
65 '\u2019': '',
66 '\u00c2': '',
67 '\u00a2': '',
68 '\u02dc': '',
69 '\u0153': '',
70 '\u00b8': '',
71 '\u02dd': '',
72 '\u0161': '',
73 '\u2030': '',
74 '\u00CF': '',
75 '\u2021': ''
76 }
77
78 # Applica le sostituzioni
79 for old, new in replacements.items():
80 text = text.replace(old, new)
81
82 # Rimuove i pattern specifici
83 text = re.sub(r'\‍(⋅\‍)', '', text)
84
85 # Rimuove le formule matematiche tra $ $
86 text = re.sub(r'\$[^\$]*\$', '', text)
87
88 # Rimuove spazi multipli
89 text = re.sub(r'\s+', ' ', text)
90
91 # Rimuove parentesi vuote
92 text = re.sub(r'\‍(\s*\‍)', '', text)
93
94 return text.strip()

96def remove_control_characters(text):
97 """!
98 @brief Remove invalid control characters from text
99 @param text Input text potentially containing control characters
100 @return Text with control characters removed
101 @details Removes ASCII control characters (0x00-0x1F and 0x7F) that can
102 cause issues in text processing and database storage
103 """
104 if not isinstance(text, str):
105 return text
106
107 # Rimuove i caratteri di controllo (da 0x00 a 0x1F e 0x7F)
108 return re.sub(r'[\x00-\x1F\x7F]', '', text)
