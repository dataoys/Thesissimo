19def calculate_precision_recall(results, relevant_docs):
20 """!
21 @brief Calculate precision and recall metrics for search results
22 @param results List of search results from Whoosh engine
23 @param relevant_docs List of relevant document IDs for evaluation
24 @return Tuple containing (precision, recall) values
25 @details Computes precision as true positives / retrieved documents
26 and recall as true positives / total relevant documents
27 """
28 retrieved_docs = len(results)
29 true_positives = sum(1 for doc in results if doc[0] in relevant_docs)
30
31 precision = true_positives / retrieved_docs if retrieved_docs > 0 else 0
32 recall = true_positives / len(relevant_docs) if relevant_docs else 0
33
34 return precision, recall
