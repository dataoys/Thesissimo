60def scraping(url):
61 """!
62 @brief Scrape document content from a single URL
63 @param url The URL to scrape for document content
64 @return Dictionary containing extracted document fields or None if failed
65 @details Extracts title, abstract, corpus, keywords, and URL from academic papers.
66 Handles LaTeX content, implements error recovery, and respects rate limiting.
67 """
68 try:
69 print(f"Scraping URL: {url}")
70 time.sleep(2) # Pausa tra le richieste
71
72 # Controllo per la pausa
73 pause_event.wait() # Questo blocca il thread fino a quando non è stato rilasciato il flag di pausa
74
75 headers = {
76 'User-Agent': get_random_user_agent(),
77 'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
78 'Accept-Language': 'en-US,en;q=0.5',
79 'Accept-Encoding': 'gzip, deflate, br',
80 'Connection': 'keep-alive',
81 'Upgrade-Insecure-Requests': '1',
82 'Cache-Control': 'no-cache',
83 'Pragma': 'no-cache',
84 'DNT': '1',
85 'Sec-GPC': '1'
86 }
87
88 response = requests.get(
89 url,
90 headers=headers,
91 timeout=30,
92 allow_redirects=True
93 )
94
95 if response.status_code == 403:
96 raise Exception("Access forbidden - waiting longer before retry")
97
98 response.raise_for_status()
99 response.encoding = response.apparent_encoding
100
101 soup = BeautifulSoup(response.text, 'html.parser')
102
103 # Estrai il titolo
104 h1_tag = soup.find('h1')
105 if not h1_tag:
106 return None
107
108 titolo = h1_tag.text.strip()
109
110 # Estrazione dell'abstract
111 page_div = soup.find('div', class_='ltx_abstract')
112 abstract = ""
113 if page_div:
114 abstract = page_div.get_text(strip=True)
115
116 # Corpo del contenuto
117 corpo = " ".join([p.text.strip() for p in soup.find_all('p')])
118 corpo = cleanText(corpo)
119
120 # Estrazione delle parole chiave
121 ltx_keywords = soup.find('div', class_='ltx_keywords')
122 if ltx_keywords is None:
123 ltx_keywords = soup.find('div', class_='ltx_classification')
124
125 keywords = ltx_keywords.text.strip() if ltx_keywords else ''
126
127 result = {
128 'title': titolo,
129 'abstract': abstract,
130 'corpus': corpo,
131 'keywords': keywords,
132 'url': url
133 }
134
135 return result
136
137 except Exception as e:
138 print("Link non trovato, passiamo al prossimo")
139 return None

141def process_urls_sequential(urls):
142 """!
143 @brief Process list of URLs sequentially with progress tracking
144 @param urls List of URLs to scrape
145 @return List of successfully scraped document dictionaries
146 @details Implements sequential processing with progress bar and error handling.
147 Respects pause events and provides real-time feedback.
148 """
149 results = []
150
151 for url in tqdm(urls):
152 result = scraping(url)
153 if result:
154 print(f"URL nel risultato: {result['url']}")
155 results.append(result)
156
157 return results

159def monitor_input():
160 """!
161 @brief Monitor user input for pause/resume commands during scraping
162 @details Runs in separate thread to handle real-time pause/resume functionality.
163 Accepts 'pause' and 'resume' commands to control scraping process.
164 @return None
165 """
166 global pause_event

172 elif user_input == "resume":
173 print("Resuming scraping...")
174 pause_event.set() # Riprende lo scraping

176def init():
177 """!
178 @brief Initialize and orchestrate the complete web scraping pipeline
179 @details Main function that coordinates the entire scraping workflow:
180 1. Creates necessary directories and files
181 2. Generates URLs for scraping
182 3. Starts monitoring thread for user commands
183 4. Executes sequential scraping with progress tracking
184 5. Processes and cleans scraped documents
185 6. Imports cleaned data into PostgreSQL database
186 @return None
187 @throws Exception if directory creation, file I/O, or database operations fail
188 """
189 # Crea le directory se non esistono
190 results_dir = project_root / "WebScraping/results"
191 results_dir.mkdir(parents=True, exist_ok=True)
192
193 # Crea il file Docs.json se non esiste
194 if not Path(NOME_FILE).exists():
195 with open(NOME_FILE, 'w', encoding='utf-8') as f:
196 json.dump([], f)
197
198 urls = list(UrlGenerators())
199 print(f"Inizio scraping di {len(urls)} URL...")
200 start_time = time.time()
201
202 # Inizia il thread di monitoraggio dell'input
203 input_thread = threading.Thread(target=monitor_input, daemon=True)
204 input_thread.start()
205
206 # Usa scraping sequenziale
207 results = process_urls_sequential(urls)
208
209 addToJson(results, NOME_FILE)
210
211 end_time = time.time()
212 print(f"Scraping completato in {end_time - start_time:.2f} secondi")
213
214 cleaned_docs = clean_documents_in_batches(NOME_FILE,"WebScraping/results/Docs_cleaned.json", 1000)
215
216 with open(FILE_PULITO, 'w', encoding='utf-8') as f:
217 json.dump(cleaned_docs, f, indent=4, ensure_ascii=False)
218
219 resetTable()
220 jsonToPG(FILE_PULITO)
221
222if __name__ == '__main__':
223 init()
