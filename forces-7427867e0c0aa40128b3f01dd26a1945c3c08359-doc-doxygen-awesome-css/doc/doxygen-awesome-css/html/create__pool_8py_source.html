82def load_queries_from_uin(uin_file_path):
83 """!
84 @brief Load queries from the UIN (User Information Need) file
85 @param uin_file_path Path to the uin.txt file containing test queries
86 @return List of query strings extracted from the file
87 @details Parses the uin.txt file format where queries follow "Query:" markers
88 on subsequent lines. Handles encoding and file not found errors.
89 """
90 queries = []
91 try:
92 with open(uin_file_path, 'r', encoding='utf-8') as f:
93 lines = f.readlines()
94 i = 0
95 while i < len(lines):
96 line = lines[i].strip()
97 if line.startswith("Query:"):
98 if i + 1 < len(lines):
99 query_text = lines[i+1].strip()
100 if query_text:
101 queries.append(query_text)
102 i += 1
103 i += 1
104 except FileNotFoundError:
105 print(f"ERROR: uin.txt not found at {uin_file_path}")
106 return queries

108def get_pylucene_top_n_ids(query_string, searcher, n=TOP_N_FOR_POOLING, ranking_type="BM25"):
109 """!
110 @brief Retrieve top N document IDs from PyLucene search engine
111 @param query_string The search query to execute
112 @param searcher PyLucene IndexSearcher object
113 @param n Number of top results to retrieve (default: TOP_N_FOR_POOLING)
114 @param ranking_type Ranking algorithm to use ("BM25" or "TFIDF")
115 @return List of document IDs as strings
116 @details Executes search across all fields and extracts document IDs from results.
117 Handles search errors and returns empty list on failure.
118 """
119 if not pylucene_search_documents or not searcher:
120 return []
121 try:
122 results, _, _ = pylucene_search_documents(searcher, True, True, True, query_string, ranking_type)
123 doc_ids = []
124 if results:
125 for i, hit in enumerate(results.scoreDocs):
126 if i >= n:
127 break
128 doc = searcher.storedFields().document(hit.doc)
129 doc_ids.append(str(doc.get("id")))
130 return doc_ids
131 except Exception as e:
132 print(f"Error during PyLucene search for query '{query_string}': {e}")
133 traceback.print_exc()
134 return []
